```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'
import type { ImageTask } from '../types/image'
import { TaskStatus } from '../types/image'

export const useTaskStore = defineStore('task', () => {
  const tasks = ref<ImageTask[]>([])

  function addTask(task: ImageTask) {
    tasks.value.unshift(task)  // 最新任务排在最前
  }

  function updateTask(id: string, patch: Partial<ImageTask>) {
    const idx = tasks.value.findIndex(t => t.id === id)
    if (idx !== -1) {
      tasks.value[idx] = { ...tasks.value[idx], ...patch, updatedAt: Date.now() }
    }
  }

  function cancelTask(id: string) {
    updateTask(id, { status: TaskStatus.CANCELLED })
  }

  // 按状态筛选
  const activeTasks = () => tasks.value.filter(
    t => t.status === TaskStatus.PENDING || t.status === TaskStatus.PROCESSING
  )

  return { tasks, addTask, updateTask, cancelTask, activeTasks }
})
```

```typescript
import { contextBridge, ipcRenderer } from 'electron'
import type { ImageParams } from '../types/image'
import { IPC_CHANNELS } from '../types/image'

contextBridge.exposeInMainWorld('imageApi', {
  // 提交图片任务，返回 { localId, jobId } 或抛出错误
  submit: (localId: string, params: ImageParams) =>
    ipcRenderer.invoke(IPC_CHANNELS.IMAGE_SUBMIT, localId, params),

  // 取消任务（停止轮询）
  cancel: (localId: string) =>
    ipcRenderer.invoke(IPC_CHANNELS.IMAGE_CANCEL, localId),

  // 监听主进程推送的状态更新
  onStatusUpdate: (callback: (update: { localId: string; patch: object }) => void) => {
    ipcRenderer.on(IPC_CHANNELS.IMAGE_STATUS_UPDATE, (_event, update) => callback(update))
    // 返回取消监听函数
    return () => ipcRenderer.removeAllListeners(IPC_CHANNELS.IMAGE_STATUS_UPDATE)
  }
})

// TypeScript 类型声明（供渲染进程使用）
export {}
declare global {
  interface Window {
    imageApi: {
      submit: (localId: string, params: ImageParams) => Promise<{ jobId: string }>
      cancel: (localId: string) => Promise<void>
      onStatusUpdate: (cb: (update: { localId: string; patch: object }) => void) => () => void
    }
  }
}
```

注意：如果 Phase 1 已建立 preload/index.ts，将 imageApi 部分追加到现有文件末尾，不要覆盖已有的 contextBridge 暴露内容。
