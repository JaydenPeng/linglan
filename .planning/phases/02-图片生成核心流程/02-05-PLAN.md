<template>
  <!-- 全屏遮罩 -->
  <div class="preview-overlay" @click.self="$emit('close')">
    <div class="preview-container">
      <!-- 关闭按钮 -->
      <ion-button class="close-btn" fill="clear" color="light" @click="$emit('close')">
        <ion-icon :icon="closeOutline" />
      </ion-button>

      <!-- 图片滑动区域 -->
      <ion-slides :options="slideOpts" :initial-slide="initialIndex" class="image-slides">
        <ion-slide v-for="(url, i) in task.resultUrls" :key="i">
          <img :src="url" class="preview-image" />
        </ion-slide>
      </ion-slides>

      <!-- 图片计数 -->
      <div v-if="(task.resultUrls?.length ?? 0) > 1" class="slide-counter">
        {{ currentIndex + 1 }} / {{ task.resultUrls?.length }}
      </div>

      <!-- 操作按钮栏 -->
      <div class="action-bar">
        <ion-button fill="solid" color="primary" @click="handleDownload">
          <ion-icon :icon="downloadOutline" slot="start" />
          下载
        </ion-button>
        <ion-button fill="outline" color="light" @click="showShareSheet = true">
          <ion-icon :icon="shareOutline" slot="start" />
          分享
        </ion-button>
      </div>
    </div>

    <!-- 分享选项面板 -->
    <ion-action-sheet
      :is-open="showShareSheet"
      :buttons="shareButtons"
      @did-dismiss="showShareSheet = false"
    />

    <!-- Toast 提示 -->
    <ion-toast
      :is-open="toastVisible"
      :message="toastMessage"
      :duration="2000"
      position="bottom"
      @did-dismiss="toastVisible = false"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, computed } from 'vue'
import {
  IonButton, IonIcon, IonSlides, IonSlide,
  IonActionSheet, IonToast
} from '@ionic/vue'
import { closeOutline, downloadOutline, shareOutline } from 'ionicons/icons'
import { Filesystem, Directory } from '@capacitor/filesystem'
import { Share } from '@capacitor/share'
import type { ImageTask } from '../types/image'

const props = defineProps<{ task: ImageTask; initialIndex: number }>()
const emit = defineEmits<{ close: [] }>()

const currentIndex = ref(props.initialIndex)
const showShareSheet = ref(false)
const toastVisible = ref(false)
const toastMessage = ref('')

const slideOpts = { initialSlide: props.initialIndex, speed: 300 }

const currentUrl = computed(() => props.task.resultUrls?.[currentIndex.value] ?? '')

function showToast(msg: string) {
  toastMessage.value = msg
  toastVisible.value = true
}

async function downloadImage(url: string): Promise<void> {
  // 通过 fetch 获取图片数据，转 base64 后用 Capacitor Filesystem 保存
  const response = await fetch(url)
  const blob = await response.blob()
  const base64 = await new Promise<string>((resolve) => {
    const reader = new FileReader()
    reader.onloadend = () => resolve((reader.result as string).split(',')[1])
    reader.readAsDataURL(blob)
  })
  const fileName = `linglan_${Date.now()}.jpg`
  await Filesystem.writeFile({
    path: fileName,
    data: base64,
    directory: Directory.Documents,
  })
}

async function handleDownload() {
  try {
    await downloadImage(currentUrl.value)
    showToast('已保存到相册')
  } catch (err) {
    showToast('下载失败，请重试')
  }
}

async function handleCopyLink() {
  await navigator.clipboard.writeText(currentUrl.value)
  showToast('链接已复制')
  showShareSheet.value = false
}

async function handleSystemShare() {
  await Share.share({
    title: '分享图片',
    text: props.task.params.prompt,
    url: currentUrl.value,
  })
  showShareSheet.value = false
}

const shareButtons = [
  { text: '复制图片链接', handler: handleCopyLink },
  { text: '保存到相册', handler: handleDownload },
  { text: '系统分享', handler: handleSystemShare },
  { text: '取消', role: 'cancel' },
]
</script>
